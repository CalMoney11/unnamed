document.addEventListener("DOMContentLoaded", function () { const apiUrl = "https://us-central1-ai-art-critique-460706.cloudfunctions.net/generate_art_critique"; const form = document.getElementById("critique-form"); if (!form) { console.error("critique-form not found in DOM"); return; } form.addEventListener("submit", async (e) => { e.preventDefault(); const fileInput = document.getElementById("file"); const promptInput = document.getElementById("art"); const styleInput = document.getElementById("style"); const responseDisplay = document.getElementById("response"); const formData = new FormData(); formData.append("prompt", promptInput.value); formData.append("style", styleInput.value); formData.append("file", fileInput.files[0]); responseDisplay.textContent = "Processing..."; try { const response = await fetch(apiUrl, { method: "POST", body: formData, }); const result = await response.json(); responseDisplay.textContent = result.response || result.error; } catch (err) { responseDisplay.textContent = "Something went wrong: " + err.message; } }); });
